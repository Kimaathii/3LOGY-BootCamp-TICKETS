🔁 Demo 05: Loops

1. for Loop

for (let i = 0; i < 5; i++) {
  console.log(`Count: ${i}`);
}

2. while Loop

let count = 0;
while (count < 3) {
  console.log(`Count: ${count}`);
  count++;
}

3. do...while Loop

let num = 0;
do {
  console.log(`Number: ${num}`);
  num++;
} while (num < 3);

4. for...of Loop (Arrays)

const fruits = ["apple", "banana", "orange"];

for (const fruit of fruits) {
  console.log(fruit);
}

5. break & continue

// break - exit loop
for (let i = 0; i < 10; i++) {
  if (i === 5) break;
  console.log(i);
}

// continue - skip iteration
for (let i = 0; i < 5; i++) {
  if (i === 2) continue;
  console.log(i);
}

6. Practical - Sum Numbers

function sumNumbers(n) {
  let sum = 0;
  for (let i = 1; i <= n; i++) {
    sum += i;
  }
  return sum;
}

console.log(sumNumbers(10)); // 55
