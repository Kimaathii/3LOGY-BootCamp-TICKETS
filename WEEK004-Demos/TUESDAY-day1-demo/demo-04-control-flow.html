🔀 Demo 04: Control Flow

1. if/else Statements

const age = 20;
if (age >= 18) {
  console.log("You can vote!");
} else {
  console.log("Too young to vote");
}

2. else if Chains

const score = 85;
let grade;

if (score >= 90) {
  grade = "A";
} else if (score >= 80) {
  grade = "B";
} else if (score >= 70) {
  grade = "C";
} else {
  grade = "F";
}

3. switch Statement

const day = "Monday";

switch (day) {
  case "Monday":
    message = "Start of the week!";
    break;
  case "Friday":
    message = "Almost weekend!";
    break;
  default:
    message = "Regular day";
}

4. Ternary Operator (? :)

const age = 20;
const canVote = age >= 18 ? "Yes" : "No";

5. Practical Example - Authentication

function checkAccess(username, password, is2FAVerified) {
  if (!username || !password) {
    return "Missing credentials";
  }
  
  if (username === "admin" && password === "secret") {
    if (is2FAVerified) {
      return "Access granted: Admin";
    } else {
      return "2FA required";
    }
  }
  
  return "Access denied";
}
